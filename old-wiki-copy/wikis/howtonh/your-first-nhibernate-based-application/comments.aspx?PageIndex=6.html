For those struggling to use examples like the one above with NHibernate 2.1 Alpha3, I offer this advice, which you can use or discard as you see fit:

1. Save some grief by using Fluent NHibernate.

2. Use the free SQL Server Express or SQL Server Deveoper Edition.  I use SQL Server 2008 SP1 Developer Edition.

3. Put your connection string information into an app.config file.

Example code that I use to run my modified edition of the Fluent NHibernate Examples.FirstProject:

app.config contains:

<?xml version="1.0" encoding="utf-8" ?>

<configuration>

 <connectionStrings>

   <add name="ConnectionString"

        connectionString="Server=.;database=HelloFluentNHibernate;Integrated Security=true;"

        providerName="System.Data.SqlClient" />

 </connectionStrings>

</configuration>

partial program.cs showing modifications:

       /// <summary>

       /// Sets up the required NHibernate session factory for SQL Server 2005

       /// </summary>

       /// <returns>ISessionFactory instance</returns>

       /// <remarks>

       /// Assumes there is an App.Config file that contains an appropriately

       /// defined connectionStrings section with a key named ConnectionString.

       /// </remarks>

       private static ISessionFactory CreateSessionFactory()

       {

           ISessionFactory returnValue = null;

           // Use Fluent NHibernate instead of an NHibernate XML config file

           returnValue = Fluently.Configure()

               .Database(MsSqlConfiguration.MsSql2005 // in my copy I added MsSql2008

                   .ConnectionString(c => c

                       .FromConnectionStringWithKey("ConnectionString")

                       )

                   .ShowSql()

                   .DefaultSchema("dbo")

// *** HERE'S THE ANSWER TO ONE OF THE PREVIOUS QUESTIONERS

                   .ProxyFactoryFactory("NHibernate.ByteCode.Castle.ProxyFactoryFactory, NHibernate.ByteCode.Castle")

               )

               .Mappings(m => m

                 .FluentMappings.AddFromAssemblyOf<Program>()

               )

               .ExposeConfiguration(BuildSchema)

               .BuildSessionFactory()

               ;

           return returnValue;

       }

       private static void BuildSchema(Configuration config)

       {

           // place to capture the database schema creation script

           var outputSchemaFilename = Path.Combine(System.Environment.GetFolderPath(System.Environment.SpecialFolder.MyDocuments), "Schema.sql");

           Console.WriteLine(outputSchemaFilename);

           if (File.Exists(outputSchemaFilename))

           {

               File.Delete(outputSchemaFilename);

           }

           // this NHibernate tool takes a configuration (with mapping info in)

           // and exports a database schema from it

           new SchemaExport(config)

               .SetOutputFile(outputSchemaFilename)

               // don't emit on console and for goodness sake don't execute the script or it will destroy the existing database!!!

               .Create(false, false)

               ;

       }
